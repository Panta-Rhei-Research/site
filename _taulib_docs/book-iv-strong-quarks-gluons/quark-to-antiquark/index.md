def quark_to_antiquark (q : QuarkMode) : AntiquarkMode where
  gamma_winding := -q.gamma_winding
  eta_winding := -q.eta_winding
  anti_type := match q.quark_type with
    | .up => .down
    | .down => .up
  generation := q.generation
