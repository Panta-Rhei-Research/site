```lean
def abcd_roundtrip_check (bound : TauIdx) : Bool :=
  go 2 (bound + 1)
where
  go (x fuel : Nat) : Bool :=
    if fuel = 0 then true
    else if x > bound then true
    else
      let p := from_tau_idx x
      let rt := to_tau_idx p
      (rt == x) && go (x + 1) (fuel - 1)
  termination_by fuel
```
