private theorem sp_polarity :
    ∀ s : Sector,
    (sector_physics s).polarity = match s with
      | .D => PolaritySign.ChiPlus | .A => .Balanced
      | .B => .ChiPlus | .C => .ChiMinus | .Omega => .Crossing := by
  intro s; cases s <;> rfl
