```lean
def crt_goldbach_local_solvable (n k : Nat) : Bool :=
  let p := nth_prime k
  if p < 3 then true
  else go 1 p
where
  go (r fuel : Nat) : Bool :=
    if fuel = 0 then false
    else if r >= nth_prime k then false
    else
      let p := nth_prime k
      let nr := (n + p - r) % p
      if r % p != 0 && nr != 0 then true
      else go (r + 1) (fuel - 1)
  termination_by fuel
```
