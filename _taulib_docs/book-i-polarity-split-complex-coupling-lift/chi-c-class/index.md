```lean
theorem chi_C_class (B_class : Nat → Bool) (p : Nat)
    (h_ge : 3 ≤ p) (h_B : B_class p = false) :
    chi B_class p = -1 := by
  match p, h_ge with
  | n + 3, _ =>
    show (if B_class (n + 3) then 1 else -1) = -1
    rw [h_B]
    rfl
```
