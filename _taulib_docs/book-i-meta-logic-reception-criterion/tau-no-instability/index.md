```lean
theorem tau_no_instability :
    ¬ ∃ (si : StructuralInstability), si.resonance = tau_resonance := by
  intro ⟨si, h⟩
  have hfull := si.full_resonance
  rw [h] at hfull
  simp [DiagonalResonance.isFullResonance, tau_resonance] at hfull
```
