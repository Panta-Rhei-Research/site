```lean
def between_mono_y (x bound db stages : TauIdx) : Bool :=
  go 2 (bound + 1)
where
  go (y fuel : Nat) : Bool :=
    if fuel = 0 then true
    else if y > bound then true
    else
      between_mono_z x y bound db stages && go (y + 1) (fuel - 1)
  termination_by fuel
```
