```lean
def between_mono_stages (x y z db stages : TauIdx) : Bool :=
  go 1 (stages + 1)
where
  go (k fuel : Nat) : Bool :=
    if fuel = 0 then true
    else if k > stages then true
    else
      let rx := reduce x k
      let ry := reduce y k
      let rz := reduce z k
      -- At stage k, either all collapse (trivial) or betweenness holds
      let ok := (rx == ry && ry == rz) || between rx ry rz db
      ok && go (k + 1) (fuel - 1)
  termination_by fuel
```
