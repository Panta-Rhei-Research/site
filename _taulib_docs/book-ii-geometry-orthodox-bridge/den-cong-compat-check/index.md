```lean
def den_cong_compat_check (db : TauIdx) : Bool :=
  let quads := [(2,3,4,5), (3,5,7,9), (2,4,6,8), (5,7,3,11),
                (2,6,4,12), (3,9,5,15), (7,11,2,8)]
  quads.all fun (a, b, c, d) =>
    !congruent a b c d db || cong_compat_stages a b c d db
```
