```lean
def indep_witness (k pi1 pi2 : TauIdx) : Bool :=
  go 0 (primorial k)
where
  go (x fuel : Nat) : Bool :=
    if fuel = 0 then false
    else if x >= primorial k then false
    else
      let g1 := cylinder_gen_indicator k pi1 0 x
      let g2 := cylinder_gen_indicator k pi2 0 x
      if g1 && g2 then true
      else go (x + 1) (fuel - 1)
  termination_by fuel
```
