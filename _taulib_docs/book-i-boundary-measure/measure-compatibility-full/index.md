```lean
def measure_compatibility_full (tms : TowerMeasurableSet) (db : Nat) : Bool :=
  go 1 (db + 1)
where
  go (k fuel : Nat) : Bool :=
    if fuel = 0 then true
    else if k >= db then true
    else measure_compatibility_check tms k && go (k + 1) (fuel - 1)
  termination_by fuel
```
