```lean
theorem equiv_of_int_eq (a b : TauInt)
    (h : (a.pos : Int) - (a.neg : Int) = (b.pos : Int) - (b.neg : Int)) :
    TauInt.equiv a b := by
  show a.pos + b.neg = b.pos + a.neg
  simp only [TauIdx] at *; omega
```
