```lean
theorem TauInt.equiv_symm {a b : TauInt} (h : TauInt.equiv a b) :
    TauInt.equiv b a := by
  show b.pos + a.neg = a.pos + b.neg
  have h' : a.pos + b.neg = b.pos + a.neg := h
  simp only [TauIdx] at *; omega
```
