```lean
theorem channel_dictionary_injective (t1 t2 : Truth4)
    (h : truth4ToChannel t1 = truth4ToChannel t2) :
    t1 = t2 := by
  cases t1 <;> cases t2 <;> first | rfl | (simp [truth4ToChannel] at h)
```
