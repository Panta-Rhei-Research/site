```lean
private theorem dvd_of_mod_eq_zero {a m : Nat} (h : a % m = 0) : m ∣ a := by
  have := Nat.div_add_mod a m; rw [h, Nat.add_zero] at this
  exact ⟨a / m, this.symm⟩
```
