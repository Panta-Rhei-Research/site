```lean
theorem bndlift_coherent (x : TauIdx) {j k : TauIdx} (hjk : j ≤ k) :
    reduce (bndlift x k) j = reduce x j := by
  simp only [bndlift, reduce]
  exact mod_mod_of_dvd x (primorial j) (primorial (k + 1))
    (primorial_dvd (Nat.le_succ_of_le hjk))
```
