```lean
def exists_separator (x y db : Nat) : Bool :=
  go x y 1 (db + 1)
where
  go (x y k fuel : Nat) : Bool :=
    if fuel = 0 then false
    else if k > db then false
    else if reduce x k != reduce y k then true
    else go x y (k + 1) (fuel - 1)
  termination_by fuel
```
