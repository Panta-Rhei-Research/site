```lean
theorem heyting_maximality (a b c : Truth4)
    (h : Truth4.le (Truth4.meet a c) b = true) :
    Truth4.le c (Truth4.heyting_impl a b) = true := by
  cases a <;> cases b <;> cases c <;>
    simp_all [Truth4.meet, Truth4.le, Truth4.heyting_impl]
```
