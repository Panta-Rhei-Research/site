def BHBipolarity.imbalance (bp : BHBipolarity) : PolarityImbalance where
  numer := (bp.chi_plus : Int) - (bp.chi_minus : Int)
  denom := bp.chi_plus + bp.chi_minus
  denom_pos := Nat.add_pos_left bp.plus_pos _
