```lean
theorem to_sectors_injective (a b : SplitComplex)
    (h : to_sectors a = to_sectors b) : a = b := by
  simp only [to_sectors, SectorPair.mk.injEq] at h
  obtain ⟨h1, h2⟩ := h
  -- h1: a.re + a.im = b.re + b.im
  -- h2: a.re - a.im = b.re - b.im
  ext <;> omega
```
