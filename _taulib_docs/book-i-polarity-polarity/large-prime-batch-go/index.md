```lean
def large_prime_batch_go (lo hi N : Nat) (fuel : Nat) : Bool :=
  if fuel = 0 then true
  else if lo > hi then true
  else large_prime_c_bounded lo N && large_prime_batch_go (lo + 1) hi N (fuel - 1)
termination_by fuel
```
