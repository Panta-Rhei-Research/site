```lean
def count_c_dominant_go (i n N : Nat) (acc : Nat) (fuel : Nat) : Nat :=
  if fuel = 0 then acc
  else if i > n then acc
  else
    let acc' := if is_prime_bool i && !(pol_at i N) then acc + 1 else acc
    count_c_dominant_go (i + 1) n N acc' (fuel - 1)
termination_by fuel
```
