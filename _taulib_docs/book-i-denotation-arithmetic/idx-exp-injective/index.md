theorem idx_exp_injective (n : TauIdx) (hn : n ≥ 2) :
    Function.Injective (idx_exp n) := by
  intro a b h
  simp only [idx_exp_eq_nat_pow] at h
  exact Tau.Orbit.exp_injective n hn h
