```lean
def cutSequent (s1 s2 : Sequent) (a : Formula) : Option Sequent :=
  if s1.succedent = [a] ∧ s2.antecedent.head? = some a then
    some ⟨s1.antecedent ++ s2.antecedent.tail, s2.succedent⟩
  else
    none
```
