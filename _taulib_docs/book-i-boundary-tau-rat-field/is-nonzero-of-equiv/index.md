```lean
theorem TauRat.is_nonzero_of_equiv {a b : TauRat}
    (h_equiv : TauRat.equiv a b) (h_nz : a.is_nonzero) : b.is_nonzero := by
  rw [TauRat.is_nonzero_iff_toRat_ne_zero] at *
  rw [← (equiv_iff_toRat_eq a b).mp h_equiv]
  exact h_nz
```
