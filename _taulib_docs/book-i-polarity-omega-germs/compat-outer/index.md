```lean
def compat_outer (comps : List TauIdx) (d k : TauIdx) (fuel : Nat) : Bool :=
  if fuel = 0 then true
  else if k > d then true
  else compat_inner comps k d d && compat_outer comps d (k + 1) (fuel - 1)
termination_by fuel
```
