```lean
def compat_inner (comps : List TauIdx) (k l : TauIdx) (fuel : Nat) : Bool :=
  if fuel = 0 then true
  else if k > l then true
  else
    let xk := comps.getD (k - 1) 0
    let xl := comps.getD (l - 1) 0
    (xl % primorial k == xk) && compat_inner comps (k + 1) l (fuel - 1)
termination_by fuel
```
