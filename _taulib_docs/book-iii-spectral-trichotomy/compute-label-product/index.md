```lean
def compute_label_product (label : PrimeLabel) (k : TauIdx) : TauIdx :=
  go 1 1 (k + 1)
where
  go (i acc fuel : Nat) : TauIdx :=
    if fuel = 0 then acc
    else if i > k then acc
    else
      let p := nth_prime i
      let l := label_direct p
      let acc' := if l == label then acc * p else acc
      go (i + 1) acc' (fuel - 1)
  termination_by fuel
```
