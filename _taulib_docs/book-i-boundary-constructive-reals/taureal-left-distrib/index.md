theorem taureal_left_distrib (a b c : TauReal) :
    TauReal.equiv (a.mul (b.add c)) ((a.mul b).add (a.mul c)) :=
  TauReal.equiv_of_pointwise (fun n => taurat_left_distrib (a.approx n) (b.approx n) (c.approx n))
