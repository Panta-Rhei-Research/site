```lean
theorem taureal_right_distrib (a b c : TauReal) :
    TauReal.equiv ((a.add b).mul c) ((a.mul c).add (b.mul c)) :=
  TauReal.equiv_of_pointwise (fun n => taurat_right_distrib (a.approx n) (b.approx n) (c.approx n))
```
