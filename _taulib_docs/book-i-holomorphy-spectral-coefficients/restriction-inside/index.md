theorem restriction_inside (f : StageFun) (inK : TauIdx → Bool)
    (n k : TauIdx) (hn : inK n = true) :
    (restriction f inK).b_fun n k = 0 ∧
    (restriction f inK).c_fun n k = 0 := by
  simp [restriction, hn]
