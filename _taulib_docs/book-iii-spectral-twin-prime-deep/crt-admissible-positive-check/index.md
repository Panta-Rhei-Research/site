```lean
def crt_admissible_positive_check (db : Nat) : Bool :=
  go 1 (db + 1)
where
  go (k fuel : Nat) : Bool :=
    if fuel = 0 then true
    else if k > db then true
    else
      (crt_twin_admissible k > 0) && go (k + 1) (fuel - 1)
  termination_by fuel
```
