```lean
def divisible_by_small_prime (n d : Nat) : Bool :=
  go 1 (d + 1)
where
  go (i fuel : Nat) : Bool :=
    if fuel = 0 then false
    else if i > d then false
    else
      let p := nth_prime i
      if p > 0 && n % p == 0 then true
      else go (i + 1) (fuel - 1)
  termination_by fuel
```
