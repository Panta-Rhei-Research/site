```lean
theorem count_primes_in_step_prime (lo hi : Nat) (hlt : lo < hi)
    (hp : is_prime_bool (lo + 1) = true) :
    count_primes_in lo hi = 1 + count_primes_in (lo + 1) hi := by
  rw [count_primes_in_unfold]
  simp only [show ¬(lo ≥ hi) by omega, ↓reduceIte, hp]
```
