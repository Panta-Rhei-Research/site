```lean
theorem reduction_compat (x : TauIdx) {k l : TauIdx} (h : k ≤ l) :
    reduce (reduce x l) k = reduce x k := by
  simp only [reduce]
  exact mod_mod_of_dvd x (primorial k) (primorial l) (primorial_dvd h)
```
