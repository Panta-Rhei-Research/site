```lean
private theorem mul_add_mod (a k r : Nat) : (a * k + r) % a = r % a := by
  induction k with
  | zero => simp
  | succ k' ih =>
    have h_rearr : a * (k' + 1) + r = (a * k' + r) + a := by
      rw [Nat.mul_succ]; omega
    rw [h_rearr, Nat.add_mod_right, ih]
```
