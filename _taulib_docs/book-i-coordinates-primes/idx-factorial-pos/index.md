```lean
theorem idx_factorial_pos (n : TauIdx) : idx_factorial n > 0 := by
  induction n with
  | zero => simp [idx_factorial]
  | succ n ih => simp only [idx_factorial]; exact Nat.mul_pos (Nat.succ_pos n) ih
```
