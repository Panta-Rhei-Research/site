```lean
theorem euclid_lemma {p a b : TauIdx} (hp : idx_prime p) (h : p ∣ a * b) :
    p ∣ a ∨ p ∣ b := by
  rcases Classical.em (p ∣ a) with hpa | hpa
  · exact Or.inl hpa
  · exact Or.inr (coprime_dvd_of_dvd_mul (prime_coprime_of_not_dvd hp hpa) h)
```
