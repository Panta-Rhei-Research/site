theorem TauRat.inv_mul_cancel (q : TauRat) (h : q.is_nonzero) :
    TauRat.equiv ((q.inv h).mul q) TauRat.one := by
  rw [equiv_iff_toRat_eq, toRat_mul, toRat_inv, toRat_one]
  rw [(TauRat.is_nonzero_iff_toRat_ne_zero q).mp h
      |> inv_mul_cancel₀]
