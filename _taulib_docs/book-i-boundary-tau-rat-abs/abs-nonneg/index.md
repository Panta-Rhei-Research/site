theorem TauRat.abs_nonneg (q : TauRat) : 0 ≤ q.abs.toRat := by
  by_cases h : 0 ≤ q.toRat
  · rw [toRat_abs_of_nonneg h]; exact h
  · push_neg at h
    rw [toRat_abs_of_neg h]
    linarith
