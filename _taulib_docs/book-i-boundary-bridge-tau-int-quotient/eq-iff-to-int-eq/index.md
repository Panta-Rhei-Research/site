```lean
theorem TauIntQ.eq_iff_toInt_eq (x y : TauIntQ) :
    x = y ↔ x.toInt = y.toInt := by
  refine Quotient.inductionOn₂ x y (fun a b => ?_)
  show (a.toQ = b.toQ) ↔ (a.toInt = b.toInt)
  rw [TauInt.toQ_eq_iff, equiv_iff_toInt_eq]
```
