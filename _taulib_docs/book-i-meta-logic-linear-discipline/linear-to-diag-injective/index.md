theorem linear_to_diag_injective (l1 l2 : LinearAspect)
    (h : linear_to_diag l1 = linear_to_diag l2) : l1 = l2 := by
  cases l1 <;> cases l2 <;> simp [linear_to_diag] at h <;> rfl
