```lean
theorem compatibility_with_algebraic (f : StageFun) :
    TowerCoherent f ↔
    (∀ n k l : TauIdx, k ≤ l → reduce (f.b_fun n l) k = f.b_fun n k) ∧
    (∀ n k l : TauIdx, k ≤ l → reduce (f.c_fun n l) k = f.c_fun n k) :=
  algebraic_core.bi_square_char f
```
