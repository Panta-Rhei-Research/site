```lean
theorem root_divides_power (n : Nat) (z m : TauIdx) (k : Nat) (hk : k > 0)
    (hz : IsRootOfUnity n z m) : IsRootOfUnity (n * k) z m := by
  rw [Nat.mul_comm]
  exact root_of_unity_pow n z m k hz
```
