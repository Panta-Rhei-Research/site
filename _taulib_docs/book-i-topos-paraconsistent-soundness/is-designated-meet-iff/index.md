theorem isDesignated_meet_iff (a b : Truth4) :
    IsDesignated (Truth4.meet a b) ↔ IsDesignated a ∧ IsDesignated b := by
  cases a <;> cases b <;> simp [IsDesignated, Truth4.meet]
